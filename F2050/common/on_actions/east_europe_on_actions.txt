on_actions = {
	on_startup = {
		effect = {
			ROM = {
				country_event = { 
					id = romania_revolt.1
					days = 399
				}
			}
		}
	}
	on_civil_war_end = {
		effect = {
			if = {
				limit = {
					ROOT = { original_tag = RSL }
					from = {
						original_tag = SLO
					}
				}
				SOV = {
					add_to_faction = RSL
					set_autonomy = {
						target = RSL
						autonomy_state = autonomy_puppet
					}
				}
			}
		}
	}
	on_capitulation_immediate = {
		effect = { 
			if = {
				limit = {
					ROOT = {
						original_tag = EST
						has_war_with = SOV
					}
					FROM = { original_tag = SOV }
				}
				every_state = {
					limit = {
						is_core_of = SOV
					}
					transfer_state_to = SOV
				}
				SOV ={
					annex_country = {
						target = EST
					}
					white_peace = EST
				}	
			}
		}
	}
}
