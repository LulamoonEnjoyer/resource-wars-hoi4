			}
		}
	}

		effect = {
			if = {
				limit = {
					ROOT = { original_tag = RSL }

					add_to_faction = RSL

			}
		}
	}

			if = {
				limit = {
					ROOT = {
						original_tag = EST

				}
				every_state = {
					limit = {
						is_core_of = SOV

					transfer_state_to = SOV
				}

					annex_country = {
						target = EST
					}
					white_peace = EST
				}	
			}
		}
	}

		effect = { 
			if = {
				limit = {

						}

				}
				every_state = {
					limit = {
						is_core_of = ROM
					}
					add_core_of = BUL
					remove_core_of = ROM
				}
				every_country = {
					limit = {
						has_war_with = ROM
						is_in_faction_with = SOV
					}
					white_peace = ROM
				}

			}
